JavaScript

String Split

 	var name = 'Beverly';

 	name.split(); => ["Beverly"]
 	
 	name.split(''); => ["B", "e", "v", "e", "r", "l", "y"]

	var job = ['P','r','o','g','r','a','m','m','e','r'];

	job.join; => "Programmer"

	var array = [5,4,7,9];

	for (var i in array){

	}
